set terminal pdfcairo
set output 'filters_distributions.pdf'
set datafile separator ","
set key off

#Box
set title 'Box Filter - x component'
set xlabel 'Distance from pixel centre'
set ylabel 'Weight'
set autoscale x
set yrange [0:1.2]
set xtics 0.5
set ytics 0,0.5,1
plot path."data_box_x.csv" with lines
set title 'Box Filter - y component'
plot path."data_box_y.csv" with lines
set title 'Box Filter - both components'
set xlabel 'Distance from centre - x'
set ylabel 'Distance from centre - y'
set zlabel 'Weight'
set autoscale xy
set hidden3d
set isosamples 60
set ticslevel 0
set xtics 0.5
set ytics 0.5
set ztics 0,0.5,1
set zrange [0:1]
splot path."data_box_xy.csv" with lines
